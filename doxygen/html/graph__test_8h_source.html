80  vertex_t u = *ui;

233  void test_add_edge(vertex_t u, vertex_t v, Graph& g)
234  {
235  Graph cpy;
236  std::vector<vertex_t> iso_vec(num_vertices(g));
237  IsoMap iso_map(iso_vec.begin(), get(vertex_index, g));
238  copy_graph(g, cpy, orig_to_copy(iso_map));
239 
240  bool parallel_edge_exists = container_contains(adjacent_vertices(u, g), v);
241 
242  std::pair<edge_t, bool> p = add_edge(u, v, g);
243  edge_t e = p.first;
244  bool added = p.second;
245 
246  if (is_undirected(g) && u == v) // self edge
247  BOOST_CHECK(added == false);
248  else if (parallel_edge_exists)
249  BOOST_CHECK(allows_parallel_edges(g) && added == true
250  || !allows_parallel_edges(g) && added == false);
251  else
252  BOOST_CHECK(added == true);
253 
254  if (p.second == true) { // edge added
255  BOOST_CHECK(num_edges(g) == num_edges(cpy) + 1);
256 
257  BOOST_CHECK(container_contains(out_edges(u, g), e) == true);
258 
259  BOOST_CHECK((verify_isomorphism
260  (make_filtered_graph(g, ignore_edge(e)), cpy, iso_map)));
261  }
262  else { // edge not added
263  if (! (is_undirected(g) && u == v)) {
264  // e should be a parallel edge
265  BOOST_CHECK(source(e, g) == u);
266  BOOST_CHECK(target(e, g) == v);
267  }
268  // The graph should not be changed.
269  BOOST_CHECK((verify_isomorphism(g, cpy, iso_map)));
270  }
271  } // test_add_edge()
