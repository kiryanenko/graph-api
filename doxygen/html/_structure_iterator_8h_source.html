38  } else {
39  _pair = _s->ngr(key - 1);
40  }
41  if (_pair.status == ERR || (end > data_t(0) && _pair.key > end)) {
42  _pair.key = _end;
43  _pair.status = ERR;
44  _pair.value = 0;
45  return;
46  }
47  }
48  }

56  if (other._pair.status == INIT_STATUS) *other;
57  return (_pair.status == ERR && other._pair.status == ERR) ||
