245  template<int I>
246  class AdjacentEdgesIterator :
247  public iterator_facade<
248  AdjacentEdgesIterator<I>,
249  edge_descriptor,
250  bidirectional_traversal_tag,
251  edge_descriptor>
252  {
253  friend class iterator_core_access;
254 
255  const SpuUltraGraph *_g;
256  vertex_descriptor _v;
257  edge_descriptor _edge;
258 
259  public:
260  AdjacentEdgesIterator() : _g(nullptr), _v(0), _edge(0) {}
261  AdjacentEdgesIterator(const SpuUltraGraph *g, vertex_descriptor v, edge_descriptor edge=0) : _g(g), _v(v), _edge(edge) {}
262  edge_descriptor dereference() const { return _edge; }
263  bool equal(const AdjacentEdgesIterator& other) const { return _edge == other._edge; }
264  void increment() {
265  auto key = _g->vertex_key(_v, I, _edge);
266  auto resp = _g->_vertex_struct.ngr(key);
267  key = resp.key;
268  if (resp.status == ERR || (id_t) key[VERTEX_ID] != _v || (int) key[INCIDENCE] != I || !_g->is_edge_id_valid(key[EDGE_ID])) {
269  _edge = _g->max_edge_id() + 1;
270  return;
271  }
272  _edge = key[EDGE_ID];
273  }
274  void decrement() {
275  auto key = _g->vertex_key(_v, I, _edge);
276  auto resp = _g->_vertex_struct.nsm(key);
277  key = resp.key;
278  if (resp.status == ERR || (id_t) key[VERTEX_ID] != _v || (int) key[INCIDENCE] != I || !_g->is_edge_id_valid(key[EDGE_ID])) {
279  _edge = 0;
280  return;
281  }
282  _edge = key[EDGE_ID];
283  }
284  };

333  vertex_descriptor dereference() const { return _v; }
334  bool equal(const EdgeVerticesIterator<I>& other) const { return _v == other._v && _e == other._e; }

336  void increment() {
337  auto key = _g->edge_key(_e, I, _v);
338  auto resp = _g->_edge_struct.ngr(key);
339  key = resp.key;
340  auto vertex_id = key[VERTEX_ID];

342  _v = _g->max_vertex_id() + 1;
343  return;
344  }
345  _v = vertex_id;
346  }

348  void decrement() {
349  auto key = _g->edge_key(_e, I, _v);
350  auto resp = _g->_edge_struct.nsm(key);
351  key = resp.key;
352  auto vertex_id = key[VERTEX_ID];
353  if (!_g->is_valid_edge_resp(resp, _e, I) || !_g->is_vertex_id_valid(vertex_id)) {
