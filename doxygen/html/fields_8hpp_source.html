32 /* Fields splited data_t */
33 template <typename NameT = u8>
34 class Fields : public AbstractFields<NameT>
35 {
36 private:
37  /* Types */
38  using Length = typename AbstractFields<NameT>::Length;
39  using Data = typename AbstractFields<NameT>::Data;
40  Length length; // Length
41  Data data; // Data
42 
43 public:
44  /* Constructor */
45  Fields(Length fields_length): AbstractFields<NameT>(), length(fields_length)
46  {
47  for ( auto ex : length )
48  {
49  data.push( {ex.name, 0} );
50  }
51  }
52 
53  /* data_t transform operator */
54  operator data_t() override
55  {
56  data_t ret = 0;
57  u8 shift = 0;
58  for(auto ex : length)
59  {
60  try
61  {
62  ret = ret | ( ( data[ex.name] & Length::mask(ex.length) ) << shift );
63  }
64  catch(DidNotFoundDataByName<NameT>&) {}
65  shift += ex.length;
66  }
67  return ret;
68  }
69 
70  /* Set data_t for fields data */
71  Fields& operator= (data_t fields_data) override
72  {
73  data.clear();
74 
75  u8 shift = 0;
76  for ( auto ex : length )
77  {
78  data_t mask = Length::mask(ex.length);
79  data_t flow = ( fields_data >> shift ) & mask;
80  data.push( {ex.name, flow} );
81  shift += ex.length;
82  }
83 
84  return *this;
85  }
86 
87  /* Set Data for fields data */
88  Fields& operator= (Data fields_data) override
89  {
90  data.clear();
91 
92  for ( auto ex : fields_data )
93  {
94  data_t flow = ex.data & length.fieldMask(ex.name);
95  data.push( {ex.name, flow} );
96  }
97 
98  return *this;
99  }
100 
101  /* Subscript operator */
102  data_t& operator[](NameT name) override { return data[name]; }
103 
104  /* Friends functions used int foreach cycle */
105  friend auto begin(Fields<NameT>& fields) { return begin(fields.data); }
106  friend auto end(Fields<NameT>& fields) { return end(fields.data); }
107 
108 };
